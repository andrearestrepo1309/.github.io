analisis_ventas.sql
-- Top productos por revenue
SELECT
  p.nombre,
  SUM(o.cantidad * o.precio) AS revenue,
  RANK() OVER (
    ORDER BY SUM(o.cantidad * o.precio) DESC
  ) AS ranking
FROM ordenes o
JOIN productos p ON o.producto_id = p.id
WHERE o.fecha >= '2024-01-01'
GROUP BY p.nombre
HAVING SUM(o.cantidad) > 50
ORDER BY revenue DESC;
